 59 	 * A convenience function to convert javascript hashes into RHQ's
 60 	 * configuration objects.
 61 	 * <p>
 62 	 * The conversion of individual keys in the hash follows these rules:
 63 	 * <ol>
 64 	 * <li> if a value of a key is a javascript array, it is interpreted as
 65 	 * PropertyList
 66 	 * <li> if a value is a hash, it is interpreted as a PropertyMap
 67 	 * <li> otherwise it is interpreted as a PropertySimple
 68 	 * <li> a null or undefined value is ignored
 69 	 * </ol>
 70 	 * <p>
 71 	 * Note that the conversion isn't perfect, because the hash does not contain
 72 	 * enough information to restore the names of the list members.
 73 	 * <p>
 74 	 * Example: <br/>
 75 	 *
 76 	 * <pre><code>
 77 	 * {
 78 	 * 	simple : "value",
 79 	 * 	list : [ "value1", "value2" ],
 80 	 * 	listOfMaps : [ {
 81 	 * 		k1 : "value",
 82 	 * 		k2 : "value"
 83 	 * 	}, {
 84 	 * 		k1 : "value2",
 85 	 * 		k2 : "value2"
 86 	 * 	} ]
 87 	 * }
 88 	 * </code></pre>
 89 	 *
 90 	 * gets converted to a configuration object: Configuration:
 91 	 * <ul>
 92 	 * <li> PropertySimple(name = "simple", value = "value")
 93 	 * <li> PropertyList(name = "list")
 94 	 * <ol>
 95 	 * <li>PropertySimple(name = "list", value = "value1")
 96 	 * <li>PropertySimple(name = "list", value = "value2")
 97 	 * </ol>
 98 	 * <li> PropertyList(name = "listOfMaps")
 99 	 * <ol>
100 	 * <li> PropertyMap(name = "listOfMaps")
101 	 * <ul>
102 	 * <li>PropertySimple(name = "k1", value = "value")
103 	 * <li>PropertySimple(name = "k2", value = "value")
104 	 * </ul>
105 	 * <li> PropertyMap(name = "listOfMaps")
106 	 * <ul>
107 	 * <li>PropertySimple(name = "k1", value = "value2")
108 	 * <li>PropertySimple(name = "k2", value = "value2")
109 	 * </ul>
110 	 * </ol>
111 	 * </ul>
112 	 * Notice that the members of the list have the same name as the list itself
113 	 * which generally is not the case.

211 		while(iterator.hasNext()) {

216 					propDef = configDef.getPropertyDefinitions().get(prop.name);
217 				}
218 				else if (configDef instanceof PropertyDefinitionMap) {
219 					propDef = configDef.get(prop.name);
220 				}
221 			}
222 			(function(parent, prop) {
223 				function isArray(obj) {
224 					return typeof(obj) == 'object' && (obj instanceof Array);
225 				}
226 
227 				function isHash(obj) {
228 					return typeof(obj) == 'object' && !(obj instanceof Array);
229 				}
230 
231 				var me = arguments.callee;
232 
233 				var representation = null;
234 
235 				if (prop instanceof PropertySimple) {
236 					if (propDef && propDef instanceof PropertyDefinitionSimple) {
237 						// TODO implement all propertySimple types ..
238 						if (propDef.getType() == PropertySimpleType.BOOLEAN) {
239                             if (prop.booleanValue !=null) {
240                                 if (prop.booleanValue == false) {
241                                 	representation = false;
242                                 }
243                                 else {
244                                 	representation = true;
245                                 }
246                             }
247                         }
248 						else if (propDef.getType() == PropertySimpleType.DOUBLE
249 								|| propDef.getType() == PropertySimpleType.INTEGER
250 								|| propDef.getType() == PropertySimpleType.LONG
251 								|| propDef.getType() == PropertySimpleType.FLOAT

263

277 				}

279 				if (isArray(parent)) {
280 					parent.push(representation);
281 				} else if (isHash(parent) && !prop.name.startsWith("__")) {
282 					parent[prop.name] = representation;
283 				}
284 			})(ret, prop);
285 		}

287 
288 		})(configuration);

308 			// we only iterrate over values
309 			if (values.hasOwnProperty(k)) {
310 				// parent - parent configuration
311 				// definition - parent configuration definition
312 				// key - config key to be applied
313 				// value - value to be applied
314 				(function (parent,definition,key,value) {
315 					var propDef = null;
316 					var prop = null;
317 					// decide which type of property are we working with
318 					if (definition instanceof PropertyDefinitionMap) {
319 						// println("DEF is map");
320 						propDef = definition.get(key);
321 					} else if (definition instanceof PropertyDefinitionList) {
322 						// println("DEF is list");
323 						propDef = definition.getMemberDefinition();
324 					} else if (definition instanceof ConfigurationDefinition) {

326 						propDef = definition.getPropertyDefinitions().get(key);
327 					}
328 
329 					if (propDef==null) {
330 						common.debug("Unable to get PropertyDefinition for key="+key);
331 						return;
332 					}
333 					// process all 3 possible types
334 					if (propDef instanceof PropertyDefinitionSimple) {
335 						prop = new PropertySimple(key, null);
336 
337 						if (value!=null) {
338 							prop = new PropertySimple(key, new java.lang.String(value));
339 						}
340 							// println("it's simple! "+prop);
341 					} else if (propDef instanceof PropertyDefinitionList) {
342 						prop = new PropertyList(key);
343 						// println("it's list! "+prop);
344 						for(var i = 0; i < value.length; ++i) {
345 							arguments.callee(prop,propDef,"",value[i]);
346 						}
347 					} else if (propDef instanceof PropertyDefinitionMap) {
348 						prop = new PropertyMap(propDef.name);
349 						// println("it's map! "+prop);
350 						for (var i in value) {
351 							if (value.hasOwnProperty(i)) {
352 								arguments.callee(prop,propDef,i,value[i]);

354 						}
355 					}
356 					else {
357 						common.info("Unkonwn property definition! this is a bug");
358 						pretty.print(propDef);
359 						return
360 					}
361 					// now we update our Configuration node
362 					if (parent instanceof PropertyList) {
363 						parent.add(prop);
364 					} else {
365 						parent.put(prop);
366 					}
367 				}) (original,definition,k,values[k]);
368 			}
369 		}

462 			    	}
